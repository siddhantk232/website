#include<iostream>
#include<iomanip>
#include<math.h>
using namespace std;

float y(float x) {
    return (2*x-x*x);
}
int main() {
    float x0,xn,h,s;
    int i,n;
    cout<<"enter x0,xn,no of subintervals: "<<endl;
    cin>>x0>>xn>>n;
    cout<<fixed;
    h=(xn-x0)/n;
    s=y(x0)+y(xn);
    for(i=1;i<=n-1;i++)
        s+=2*y(x0+i*h);
    cout<<"value of integral is"<<setw(6)<<setprecision(4)<<(h/2)*s<<endl;
    return 0;
}

#include <iostream>
#include <iomanip>
using namespace std;

float df(float x, float y) {
  return x + y;
}

int main() {
  float x0, y0, h, x, x1, y1;
  cout << "Enter the values of x0, y0, h, x: ";
  cin >> x0 >> y0 >> h >> x;
  cout << fixed;

  x1 = x0; y1 = y0;

  while(1) {
    if (x1 > x) return 0;
    y1 += h * df(x1, y1);
    x1 += h;

    cout << "When x = "
      << setw(3) << setprecision(1)
      << x1 << " y = "
      << setw(4) << setprecision(2)
      << y1 << endl;
  }
}

#include<iostream>
#include<iomanip>
using namespace std;

float f(float x, float y) {
  return x + y * y;
}
int main() {
  float x0, y0, h, xn, x, y, k1, k2, k3, k4, k;
  cout << "Enter the values of x0, y0,"
    << "h, xn: ";
  cin >> x0 >> y0 >> h >> xn;

  x = x0; y = y0;
  cout << fixed;

  while(1) {
    if (x == xn) break;
    k1 = h * f(x, y);
    k2 = h * f(x + h / 2, y + k1 / 2);
    k3 = h * f(x + h / 2, y + k2 / 2);
    k4 = h * f(x + h, y + k3);
    k = (k1 + (k2 + k3) * 2 + k4) / 6;
    x+= h; y += k;
    cout << "When x = " << setprecision(4) << setw(8)
      << x << " y = " << setw(8) << y << endl;
  }
  return 0;
}

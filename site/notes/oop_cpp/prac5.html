(practical 15) Aim: WAP to demonstrate usage of a friend function. 

#include <iostream>
using namespace std;

class xyz {
  int a, b;

public:
  xyz(int i, int j) : a(i), b(j) {}
  void display() {
    cout << "a = " << a << endl;
    cout << "b = " << b << endl;
  }
  friend int sum(xyz);
};

int sum(xyz o) { return o.a + o.b; }

int main() {
  xyz x(10, 20);
  x.display();
  cout << "sum =  " << sum(x) << endl;
  return 0;
}

a = 10
b = 20
sum =  30

(practical 16) Aim: WAP to demonstrate usage of a friend class. 

#include <iostream>

using namespace std;

class Min;

class TwoValues {
  int a, b;

 public:
  TwoValues(int i, int j) : a(i), b(j) {}
  friend class Min;
};

class Min {
 public:
  int min(TwoValues t) { return t.a < t.b ? t.a : t.b; }
};

int main() {
  TwoValues t(10, 5);
  Min m;
  cout << "Minimum value is = " << m.min(t) << endl;
}

Minimum value is = 5

(practical 17) Aim: WAP to give access of a class to a method defined in another class. 

#include <iostream>

using namespace std;

class beta;

class alpha {
  int data;

public:
  alpha(int i) : data(i) {}
  int sum(beta);  // a.sum()
};

class beta {
  int data;

public:
  beta(int i) : data(i) {}
  friend int alpha::sum(beta);
};

int alpha::sum(beta b) { return data + b.data; }

int main() {
  alpha a(10);
  beta b(20);
  cout << "sum=" << a.sum(b) << endl;
  return 0;
}

sum=30

(practical 18) Aim: WAP for a friend function that take two different class as parameters. 

#include <iostream>

using namespace std;

class beta;

class alpha {
  int data;

public:
  alpha(int i) : data(i) {}
  friend int sum(alpha a, beta b);
};

class beta {
  int data;

public:
  beta(int i) : data(i) {}
  friend int sum(alpha a, beta b);
};

int sum(alpha a, beta b) { return a.data + b.data; }

int main() {
  alpha a(10);
  beta b(20);
  cout << "Sum = " << sum(a, b) << endl;
  return 0;
}

Sum = 30

#include <iostream>
#include <cmath>

using namespace std;

class Polar {
  float radius;
  float angle;

public:
  Polar(float radius, float angle) : radius(radius), angle(angle) {}

  Polar operator + (Polar rhs) {
    float x1 = radius * cos(angle);
    float y1 = radius * sin(angle);
    float x2 = rhs.radius * cos(rhs.angle);
    float y2 = rhs.radius * sin(rhs.angle);

    float x = x1 + x2;
    float y = y1 + y2;

    float r = sqrt(x*x + y * y);
    float a = atan(y / x);
    return Polar(r, a);
  }

  void show() {
    cout << radius << " radius, " << angle << " degrees" << endl;
  }
};

int main() {
  Polar p1(1.0, 45.0), p2(1.0, 45.0);

  Polar res = p1 + p2;

  res.show();
}

#include <iostream>

using namespace std;

class Number {
  double data;
public:
  Number(double data) : data(data) {}

  Number operator + (Number rhs) {
    return Number(data + rhs.data);
  }
  Number operator - (Number rhs) {
    return Number(data - rhs.data);
  }
  Number operator * (Number rhs) {
    return Number(data * rhs.data);
  }
  Number operator / (Number rhs) {
    return Number(data / rhs.data);
  }

  void show() {
    cout << data << endl;
  }
};


int main() {
  Number n1(20.0), n2(10.0);
  Number x = n1 + n2;
  x.show();
  Number y = n1 - n2;
  y.show();
  Number z = n1 * n2;
  z.show();
  Number a = n1 / n2;
  a.show();
}

#include <iostream>

class Miles {
  double miles;

public:
  Miles(double miles) : miles(miles) {}
  Miles(Kilometers k) {
    miles = k.kms() / 1.609344;
  }

  void show() {
    cout << miles << " miles." << endl;
  }

  double miles() {
    return miles;
  }

  operator Kilometers() {
    return Kilometers(miles * 1.609344);
  }
};

class Kilometers {
  double kms;

public:
  Kilometers(double kms) : kms(kms) {}
  Kilometers(Miles m) {
    kms = m.miles() * 1.609344;
  }

  void show() {
    cout << kms << " kms." << endl;
  }

  double kms() {
    return kms;
  }

  operator Miles() {
    return Miles(kms / 1.609344);
  }
};


int main() {
  Miles m(1000);
  Kilometers k(10);

  Miles x = k;
  Kilometers y = m;

  Miles a = Kilometers(20);
  Kilometers b = Miles(1000);

  x.show();
  y.show();
  a.show();
  b.show();
}

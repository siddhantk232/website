(practical 39) Aim: WAP to write buffer contents into a binary file and read that binary data. 

#include<iostream>
#include<fstream>

using namespace std;

const int MAX = 10;
int buffer[MAX];

int main() {
  for(int j = 1; j <= MAX; j++)
    buffer[j] = j;

  ofstream of("test4.dat", ios::binary);

  of.write(reinterpret_cast<char*>(buffer), MAX * sizeof(int));

  of.close();

  for(int j = 0; j <MAX; j++)
    buffer[j] = 0;

  ifstream if1("test4.dat", ios::binary);

  if1.read(reinterpret_cast<char*>(buffer), MAX * sizeof(int));

  for(int j = 0; j<MAX; j++)
    cout<<buffer[j];

  if1.close();
  return 0;
}

(practical 41) Aim: Use the generalised fstream class object to read/write binary data of an object. 

#include<iostream>
#include<fstream>
#include<string>

using namespace std;

class student {
    int rollno;
    string name;
public:
    void getdata() {
        cout<<"enter your roll no.:";
        cin>>rollno;
        cout<<"enter your name:";
        cin>>name;
    }
    void display() {
        cout<<"Roll no.:"<<rollno<<endl;
        cout<<"Name:"<<name<<endl;
    }
};

int main () {
    fstream iost;
    iost.open("text.8dat",ios::binary|ios::in|ios::out|ios::app);
    char ch;
    student s;

    do {
        cout<<"enter data for the student: \n";
        s.getdata();
        iost.write(reinterpret_cast<char*>(&s),sizeof(s));
        cout<<"do you want to enter another record? (Y/N): ";
        cin>>ch;
    } while(ch=='Y'||ch=='y');

    iost.seekg(0);

    iost.read(reinterpret_cast<char *>(&s), sizeof(s));

    while(!iost.eof()) {
        s.display();
        iost.read(reinterpret_cast<char *>(&s), sizeof(s));
    }

    iost.close();

    return 0;
}

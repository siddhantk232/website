(practical 11) Aim: WAP to demonstrate usage of copy constructor. 

#include<iostream>

using namespace std;

class Sample {
  int *i;
  int size;

  public:

  Sample() {
    size = 3;
    i = new int[size];
    for (int y = 0; y < 3; ++y) {
      i[y] = y + 1;
    }
  }
  Sample(int x): size(x) {
    i = new int[size];
    for (int y = 0; y < x; ++y) {
      i[y] = y * 2 + 1;
    }
  }

  Sample(Sample &ob) {
    i = new int[ob.size];
    size = ob.size;
    for (int y = 0; y < size; ++y) {
      i[y] = ob.i[y];
    }
  }

  void show_i() {
    for (int y = 0; y < size; ++y) {
      cout << i[y] << " ";
    }
    cout << endl;
  }
};

int main() {
  Sample s1;
  cout << "s1: ";
  s1.show_i();
  Sample s2(3);
  cout << "s2: ";
  s2.show_i();
  Sample s3 = s2;
  cout << "s3: ";
  s3.show_i();
  Sample s4(s3);
  cout << "s4: ";
  s4.show_i();
}

s1: 1 2 3
s2: 1 3 5
s3: 1 3 5
s4: 1 3 5

(practical 12) Aim: WAP to demonstrate static class attributes. 

#include<iostream>

using namespace std;

class Shared {
  int a;
  static int b;

public:
  void setData(int i, int j) {
    a = i;
    b = j;
  }

  void show() {
    cout << "non-static data member: " << a << endl;
    cout << "static data member: " << b << endl;
  }
};

int Shared::b;

int main() {
  Shared s = Shared();
  Shared x = Shared();
  s.setData(19, 20);
  x.setData(29, 30);
  s.show();
  x.show();
}

non-static data member: 19
static data member: 30
non-static data member: 29
static data member: 30

(practical 13) Aim: WAP to count class instances. 

#include<iostream>

using namespace std;

class Count {
  int id;
  static int count;

public:
  Count() {
    id = count;
    count++;
  }

  ~Count() {
    count--;
  }

  int getId() { return id; };

  static int getCount() { return count; };
};

int Count::count = 0;

int main() {
  Count *a = new Count();
  Count *b = new Count();
  Count *c = new Count();
  Count *d = new Count();

  cout << "Count class has " << Count::getCount() << " instances" << endl;
  cout << "Class ids are:" << endl;
  cout << "a: " << a->getId() << endl;
  cout << "b: " << b->getId() << endl;
  cout << "c: " << c->getId() << endl;
  cout << "d: " << d->getId() << endl;

  delete b;

  cout << "Count class has " << Count::getCount() << " instances after deletion" << endl;

  delete a;
  delete c;
  delete d;

  cout << "Count class has " << Count::getCount() << " instances after deleting all objects" << endl;
}

Count class has 4 instances
Class ids are:
a: 0
b: 1
c: 2
d: 3
Count class has 3 instances after deletion
Count class has 0 instances after deleting all objects

(practical 14) Aim: WAP to demonstrate lifetime of a static object in cpp. 

#include<iostream>

using namespace std;

class Sample {
  int id;
  static int count;
public:
  Sample() {
    id = count;
    count++;
    cout << "constructor called. id: " << id << endl;
  }
  ~Sample() {
    cout << "destructor called. id: " << id << endl;
    count--;
  }
};

int Sample::count;

void f() {
  static Sample s1;
}

int main() {
  cout << "main starts" << endl;
  Sample s0;
  f();
  Sample s2;
  cout << "main ends" << endl;
}

constructor called. id: 1
constructor called. id: 2
main ends

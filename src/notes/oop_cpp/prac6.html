(practical 20) Aim: WAP to solve the ambiguity problem in inheritance using virtual base class. 

#include <iostream>
using namespace std;

class Base{
public:
  int i;
};

class Derv1 : public virtual Base{
public:
  int j;
};

class Derv2 : public virtual Base{
public:
  int k;
};

class Derv3 : public Derv1 , public Derv2{
public:
  int sum;
};

int main(){
  Derv3 obj;
  obj.i = 10;
  obj.j = 20;
  obj.k = 30;
  obj.sum = obj.i + obj.j + obj.k;
  cout << obj.sum;
}

(practical 24) Aim: WAP to demonstrate virtual destructors. 

#include<iostream>

using namespace std;

class Base {
  int *x;
public:
  Base(int n) {
    x = new int;
    *x = n;
  }

  virtual ~Base() {
    cout << "Destructor of Base called" << endl;
    cout << "deleting x" << endl;
    delete x;
  }
};

class Derived : public Base {
  int *y;
public:
  Derived(int n): Base(n) {
    y = new int;
    *y = n;
  }

  ~Derived() {
    cout << "Destructor of Derived called" << endl;
    cout << "deleting y" << endl;
    delete y;
  }
};

int main() {
  Base *x = new Derived(20);
  delete x;
}

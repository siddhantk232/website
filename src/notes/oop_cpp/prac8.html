#include <iostream>

using namespace std;

int f1(int i, int j) {
  cout << "In f1()" << endl;
  return i * j;
}

int f2(int x, int y) {
  cout << "In f2()" << endl;
  return x + y;
}

void f3(void (*fp)()) {
  fp();
}

void f4() {
  cout << "In f1()" << endl;
}


int main() {
  int (*fp)(int, int);
  fp = f1;

  cout << fp(10, 20) << endl;

  fp = f2;
  cout << fp(10, 20) << endl;

  f3(f4);
  return 0;
}

(practical 31) Aim: WAP to demonstrate the use of the new and delete keywords. 

#include<iostream>
#include<cstring>

using namespace std;

class str {
  char *p;
public:
  str(char *s) {
    int len = strlen(s);
    p=new char[len+1];
    strcpy(p,s);
  }

  ~str() {
    delete[]p;
  }

  void display() {
    cout<<p;
  }
};

int main() {
  char *s = "object oriented programming";
  str *s1 = new str(s);

  s1->display();

  return 0;
}

(practical 32) Aim: WAP to demonstrate pointer to a class variable. 

#include<iostream>

using namespace std;

class XYZ {
public :
  int a;
public:

  void display() {
    cout << a <<endl;
  }
};

int main() {
  XYZ x;
  int XYZ::*p = &XYZ::a;
  x.*p = 100;

  x.display();

  XYZ *y = new XYZ;
  y->*p = 200;

  y->display();

  void (XYZ::*q)(void) = &XYZ::display;

  (x.*q)();
  (y->*q)();
  return 0;
}

(practical 33) Aim: WAP to demonstrate self referential classes. 

#include <iostream>
using namespace std;

struct node {
    int data;
    struct node *ptr;
};

class LinkedList {
    struct node *start;
public:
    LinkedList() {
        start = NULL;
    }

    void addnode(int val) {
        node *n = new node;
        n->data = val;
        n->ptr = start;
        start = n;
    }

    void display() {
        node *p = start;
        while (p != NULL) {
            cout << p -> data << endl;
            p = p -> ptr;
        }
    }

};

int main(){
    LinkedList ll;
    ll.addnode(10);
    ll.addnode(20);
    ll.addnode(30);
    ll.display();

    return 0;
}

(practical 34) Aim: WAP to demonstrate the use of the this pointer. 

#include <iostream>
using namespace std;

class xyz{
  int a, b;
public:
  // xyz(int x, int y){   // we want to use a and b as formal arguments
  //     a = x;
  //     b = y;
  // }

  xyz(int a, int b) {   
    this->a = a;
    this->b = b; // to solve ambiguity we use this pointer
  }

  void display() {
    cout << "a = " << this->a << endl;
    cout << "b = " << this->b << endl;
  }

  void address() {
    cout << "Address of this object = " << this << endl;
  }
};

int main(){
  xyz x(10, 20);
  x.display();
  x.address();
  xyz y(10, 20);
  y.address();
  xyz z(10, 20);
  z.address();
}

(practical 2) Aim: Write a program to check if a given number is prime. 

#include <iostream>
using namespace std;

int main() {
  int n;
  cout << "Enter number: ";
  cin >> n;

  int i = 2;
  while (i < n) {
    i++;
    if (n % i == 0) {
      break;
    }
  }

  if (n == i) {
    cout << n << " is prime." << endl;
  } else {
    cout << n << " is not prime." << endl;
  }
}

(practical 3) Aim: Write a program to create a basic calculator. 

#include <iostream>

using namespace std;

void print_prompt();
void reset_choice(char *choice);

int main() {
  char choice;
  do {
    print_prompt();
    cin >> choice;
    int a,b;
    cout << "Enter two numbers for " << choice << " operation: ";
    cin >> a >> b;
    switch(choice) {
      case '+':
        cout << "Sum is: " << a + b << endl;
        reset_choice(&choice);
        break;
      case '-':
        cout << "Difference is: " << a - b << endl;
        reset_choice(&choice);
        break;
      case '*':
        cout << "Product is: " << a * b << endl;
        reset_choice(&choice);
        break;
      case '/':
        cout << "Quotient is: " << a / b << endl;
        cout << "Remainder is: " << a % b << endl;
        reset_choice(&choice);
        break;
      default:
        cout << "Invalid operation." << endl << endl;
        reset_choice(&choice);
    }
  } while (choice == 'y' || choice == 'Y');
}

void print_prompt() {
  cout << "Enter your choice of operation: " << endl;
  cout << "+: Addition" << endl;
  cout << "-: Subtraction" << endl;
  cout << "*: Multiplication" << endl;
  cout << "/: Division" << endl;

  cout << "Choice (+ / - / * / /): ";
}

void reset_choice(char *choice) {
  cout << "Do you want to perform more operations? (y/n): ";
  cin >> *choice;
}

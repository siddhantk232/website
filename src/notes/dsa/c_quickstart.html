2. Data types and Variables

#include <stdio.h>

int main() {
    int var_name = 69;
    char a = 'a';
    // string reprsentation in C
    // char *name = "siddhant";
    // char name[] = "siddhant";
    char name[] = {'s', 'i', 'd', 'd', 'h', 'a', 'n', 't', '\0'};

    printf("%s\n", name);

    printf("%d\n", sizeof(var_name));
    printf("%d\n", sizeof(name));
    printf("%d\n", sizeof(a));
}

5. Control Statements/Structures

5.1. if/else/else if

Odd/Even program 

#include <stdio.h>

int main() {
    int n = 68;
    // modulus operator returns remainder of the division operation
    if (n % 2 == 0) {
        printf("even\n");
    } else {
        printf("odd\n");
    }
}

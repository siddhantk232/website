(practical 5) Aim: Study the bitwise operators. 

#include <iostream>
#include <climits>

using namespace std;

int main() {
    int a = 10;
    int b = 2;

    cout << (a & b) << endl;
    cout << (a | b) << endl;
    cout << (a ^ b) << endl;
    cout << (~a) << endl;

    cout << (a << 2) << endl;
    cout << (a >> 1) << endl;
}

(practical 7) Aim: Program to understand increment/decrement operator. 

#include <iostream>

using namespace std;

int main() {
    int a = 3;
    int b = a++;

    int c = 10;
    int d = ++c;

    cout << "a: " << a << endl;
    cout << "b: " << b << endl;
    cout << "c: " << c << endl;
    cout << "d: " << d << endl;

}
